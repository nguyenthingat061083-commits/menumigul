<script>
    // Toggle
    function toggle(el) {
        el.classList.toggle('active');
    }

    // Slider
    function updateVal(input, spanId) {
        document.getElementById(spanId).textContent = parseFloat(input.value).toFixed(1);
    }

    // Tabs
    document.querySelectorAll('.tab').forEach(btn => {
        btn.addEventListener('click', function() {
            document.querySelectorAll('.tab').forEach(b => b.classList.remove('active'));
            this.classList.add('active');
            document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
            document.getElementById('panel-' + this.dataset.tab).classList.add('active');
        });
    });

    // Activate
    function activate() {
        const config = {
            aimbot: document.querySelector('#panel-aim .toggle.active') ? 'ON' : 'OFF',
            fov: document.getElementById('fovVal').textContent,
            smooth: document.getElementById('smoothVal').textContent,
            hitbox: document.querySelector('#panel-aim select')?.value || 'Neck',
            esp: {
                box: document.querySelector('#panel-visual .toggle.active') ? 'ON' : 'OFF',
                name: document.querySelectorAll('#panel-visual .toggle')[1]?.classList.contains('active') ? 'ON' : 'OFF',
                health: document.querySelectorAll('#panel-visual .toggle')[2]?.classList.contains('active') ? 'ON' : 'OFF',
                distance: document.querySelectorAll('#panel-visual .toggle')[3]?.classList.contains('active') ? 'ON' : 'OFF',
                snapline: document.querySelectorAll('#panel-visual .toggle')[4]?.classList.contains('active') ? 'ON' : 'OFF'
            },
            mod: {
                fly: document.querySelector('#panel-mod .toggle')?.classList.contains('active') ? 'ON' : 'OFF',
                speed: document.querySelectorAll('#panel-mod .toggle')[1]?.classList.contains('active') ? 'ON' : 'OFF',
                nowater: document.querySelectorAll('#panel-mod .toggle')[2]?.classList.contains('active') ? 'ON' : 'OFF',
                ghost: document.querySelectorAll('#panel-mod .toggle')[3]?.classList.contains('active') ? 'ON' : 'OFF',
                norecoil: document.querySelectorAll('#panel-mod .toggle')[4]?.classList.contains('active') ? 'ON' : 'OFF',
                autofire: document.querySelectorAll('#panel-mod .toggle')[5]?.classList.contains('active') ? 'ON' : 'OFF'
            }
        };
        alert('✅ Đã kích hoạt cấu hình MiGul!\n' + JSON.stringify(config, null, 2));
        console.log('📦 MiGul Config:', config);
    }
</script>
